                        const newScores = [...t.scores];

                    }

                }));

                    if (t.id === teamId) {

                        const newScores = [...t.scores];

                        newScores[stageIdx] = points;

                        return { ...t, scores: newScores };

                    }

                    return t;

                }));

            };

                        if (field === 'name') return { ...t, name: value };

                            const newMembers = [...t.members];

                            newMembers[memberIdx] = value;

                            return { ...t, members: newMembers };

                        }

                    }

                    return t;

                }));

            };

                        </div>

                                                    type="text" 

                                                    value={m}

                                                        <div className="excel-cell rounded-xl overflow-hidden bg-slate-950/50">

                                                            <input 

                                                                type="number" 

                                                                value={team.scores[sIdx] || ''}

                                                                placeholder="0"

                                                                onChange={(e) => updateScore(team.id, sIdx, e.target.value)}

                                                            />

                                                        </div>

                                                    </td>

                                                    <td className="py-3 text-right space-x-2">

                                                        <button

'bg-yellow-500 border-yellow-500 text-slate-950 shadow-[0_0_15px_rgba(234,179,8,0.4)]' : 'border-slate-800 text-slate-500 hover:border-yellow-500/50 hover:text-yellow-500'}`}

                                                            及格 +50

                                                        </button>

                                                            onClick={() => addTacitScore(team.id, sIdx, 150)}

'bg-green-500 border-green-500 text-slate-950 shadow-[0_0_15px_rgba(34,197,94,0.4)]' : 'border-slate-800 text-slate-500 hover:border-green-500/50 hover:text-green-500'}`}

                                                </tr>

                                            ))}

                                        </tbody>

                                    </table>
